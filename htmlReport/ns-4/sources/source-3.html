 package by.bratchykau.receipt.utils;
 
 
 
 import by.bratchykau.receipt.model.Product;
 import by.bratchykau.receipt.model.Receipt;
 
 import java.io.BufferedWriter;
 import java.io.FileWriter;
 import java.io.IOException;
 
 public class ReceiptDecorator {
     private final Receipt receipt;
 
     public ReceiptDecorator(Receipt receipt) {
         this.receipt = receipt;
     }
 
     public void outputToConsole() {
         // Output the receipt to the console
         System.out.println("Receipt:");
         for (Product product : receipt.getProducts()) {
             if (isDiscount(product)) {
                 System.out.println(product.getName() + " x" + product.getQuantity() + " @ $" + product.getPrice()
                         + " = $" + product.getPrice().multiply(0.9, RoundMethod.FLOOR,0).multiplyForInt(product.getQuantity()) + " with discount");
 
             } else {
                 System.out.println(product.getName() + " x" + product.getQuantity() + " @ $" + product.getPrice()
                         + " = $" + product.getPrice().multiplyForInt(product.getQuantity()));
             }
         }
         System.out.println("Total: $" + receipt.getTotalAmount());
         if (receipt.getDiscountCard() != null) {
             System.out.println("Discount card: " + receipt.getDiscountCard().getNumber());
         }
     }
 
     public void outputToFile(String fileName) {
         try {
             BufferedWriter writer = new BufferedWriter(new FileWriter(fileName));
             writer.write("Receipt:");
             writer.newLine();
             for (Product product : receipt.getProducts()) {
                 if (isDiscount(product)) {
                     writer.write(product.getName() + " x" + product.getQuantity() + " @ $" + product.getPrice()
                             + " = $" + product.getPrice().multiply(0.9,RoundMethod.FLOOR,0).multiplyForInt(product.getQuantity()) + " with discount");
                     writer.newLine();
                 } else {
                     writer.write(product.getName() + " x" + product.getQuantity() + " @ $" + product.getPrice()
                             + " = $" + product.getPrice().multiplyForInt(product.getQuantity()));
                     writer.newLine();
 
                 }
             }
             writer.write("Total: $" + receipt.getTotalAmount());
             writer.newLine();
             if (receipt.getDiscountCard() != null) {
                 writer.write("Discount card: " + receipt.getDiscountCard().getNumber());
                 writer.newLine();
             }
             writer.close();
         } catch (IOException e) {
             e.printStackTrace();
         }
     }
 
     private static boolean isDiscount(Product product) {
         return product.getQuantity() > 5;
     }
 
 }
