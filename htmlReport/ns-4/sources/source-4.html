 package by.bratchykau.receipt.utils;
 
 public enum RoundMethod {
     FLOOR {
         double roundFun(double d) {
             return Math.floor(d);
         }
     },
     ROUND {
         double roundFun(double d) {
             return Math.round(d);
         }
     },
     CEIL {
         double roundFun(double d) {
             return Math.ceil(d);
         }
     };
 
     abstract double roundFun(double value);
 
 
     public int round(double roundedValue, RoundMethod roundMethod, int accuracy) {
         int tenPow = pow10(accuracy);
         int result = (int) roundMethod.roundFun(roundedValue / tenPow) * tenPow;
         return result;
     }
 
     private int pow10(int accuracy) {
         int result = 1;
         for (int i = 0; i < accuracy; i++) {
             result *= 10;
         }
         return result;
     }
