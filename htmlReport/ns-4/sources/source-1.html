 package by.bratchykau.receipt.utils;
 
 
 
 import by.bratchykau.receipt.exceptions.LineException;
 import by.bratchykau.receipt.by.bratchykau.receipt.utils.Constants;
 
 import java.util.regex.Pattern;
 
 public class ProductUtils {
 
     public static String getProductValid(String product) throws LineException {
         String pattern = "^[0-9]+-[0-9]+$";
         Pattern p = Pattern.compile(pattern);
         if (p.matcher(product).matches()) {
             return product;
         }
         throw new LineException(by.bratchykau.receipt.utils.Constants.WRONG_LINE + product);
     }
 
 
 }
