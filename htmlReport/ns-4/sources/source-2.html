 package by.bratchykau.receipt.utils;
 
 
 
 import by.bratchykau.receipt.Byn;
 import by.bratchykau.receipt.model.DiscountCard;
 import by.bratchykau.receipt.model.Product;
 import by.bratchykau.receipt.model.Receipt;
 
 import java.util.ArrayList;
 import java.util.List;
 
 public class ReceiptBuilder {
     private final List<Product> products = new ArrayList<>();
     private Byn totalAmount;
     private DiscountCard discountCard;
 
     public void addProduct(Product product) {
         products.add(product);
         totalAmount = product.getPrice().multiplyForInt(product.getQuantity()).add(totalAmount);
     }
 
     public void setDiscountCard(DiscountCard discountCard) {
         this.discountCard = discountCard;
     }
 
         public Receipt build() {
             Byn totalAmount = new Byn();
 
             // Calculate total amount of the receipt
             for (Product product : products) {
                 totalAmount = product.getPrice().multiplyForInt(product.getQuantity()).add(totalAmount);
             }
 
             // Apply discount on the presented card (if there is one)
             if (discountCard != null) {
                 //totalAmount -= totalAmount * discountCard.getDiscountPercentage();
                 //totalAmount = totalAmount.subtract(totalAmount.findDiscount(discountCard.getDiscountPercentage()));  //20.66 - 4.132 = 16.528
                 totalAmount = totalAmount.subtract(totalAmount.multiply((double) discountCard.getDiscountPercentage() / 100, RoundMethod.CEIL, 0));
             }
 
             // Apply 10% discount on items with a quantity of more than five
             for (Product product : products) {
                 if (product.getQuantity() > 5) {
                     totalAmount = totalAmount.subtract(totalAmount.findDiscount(10));
                 }
             }
 
             return new Receipt(products, totalAmount, discountCard);
         }
 }
