 package by.bratchykau.receipt;
 
 
 
 import by.bratchykau.receipt.exceptions.LineException;
 import by.bratchykau.receipt.Byn;
 import by.bratchykau.receipt.model.DiscountCard;
 import by.bratchykau.receipt.model.Product;
 import by.bratchykau.receipt.model.Receipt;
 import by.bratchykau.receipt.utils.ProductUtils;
 import by.bratchykau.receipt.utils.ReceiptBuilder;
 import by.bratchykau.receipt.utils.ReceiptDecorator;
 
 import java.util.ArrayList;
 import java.util.HashMap;
 import java.util.List;
 import java.util.Map;
 
 public class Runner {
     public static void main(String[] args) {
         // create products
         List<Product> products = new ArrayList<>();
         products.add(new Product(1, "Product 1", new Byn(200)));
         products.add(new Product(2, "Product 2", new Byn(200)));
         products.add(new Product(3, "Product 3", new Byn(333)));
         products.add(new Product(4, "Product 4", new Byn(400)));
         products.add(new Product(5, "Product 5", new Byn(500)));
 
 
         // create discount cards
         Map<String, DiscountCard> discountCards = new HashMap<>();
         discountCards.put("1", new DiscountCard("card1", 10));
         discountCards.put("2", new DiscountCard("card2", 20));
         discountCards.put("3", new DiscountCard("card3", 30));
         discountCards.put("1234", new DiscountCard("card-1234", 30));
 
         // parse command-line arguments
         try {
         ReceiptBuilder builder = new ReceiptBuilder();
         DiscountCard discountCard = null;
         for (String arg : args) {
             if (arg.startsWith("card-")) {
                 // this is a discount card
                 String cardNumber = arg.substring(5);
                 discountCard = discountCards.get(cardNumber);
             } else {
                 // this is a product
 
                 String validLine = ProductUtils.getProductValid(arg);
 
                 String[] parts = validLine.split("-");
 
                 int id = Integer.parseInt(parts[0]);
                 int quantity = Integer.parseInt(parts[1]);
                 Product product = new Product();
                 try {
                     product = products.get(id-1);
                 } catch (IndexOutOfBoundsException e) {
                     System.err.println(e.getMessage());
                     throw new IndexOutOfBoundsException();
                 }
                 //product = products.get(id - 1);
                 //Product product = products.get(id-1);
                 product.setQuantity(quantity);
                 if (product == null) {
                     throw new IllegalArgumentException("Invalid product id: " + id);
                 }
                 builder.addProduct(product);
             }
         }
 
         builder.setDiscountCard(discountCard);
         Receipt receipt = builder.build();
 
         // output receipt to console and file
         ReceiptDecorator decorator = new ReceiptDecorator(receipt);
         decorator.outputToConsole();
         decorator.outputToFile("src/main/resources/output.txt");
         } catch (LineException e) {
             System.err.println(e.getMessage());
         }
     }
 }
